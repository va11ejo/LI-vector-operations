127def cosseno_segunda_metade(numbers):
128 # Calcular o index da segunda metade
129 metade_index = len(numbers) // 2
130
131 # Criar um array dos numeros da segunda metade
132 segunda_metade_vector = numbers[metade_index:]
133
134 # calcular o cosseno de cada elemento da segunda metade do array
135 return [math.cos(number) for number in segunda_metade_vector]

259def check_composto(number):
260 # Números menores que 2 não são compostos
261 if number < 2:
262 return False
263
264 # Verificar se o número tem algum divisor além de 1 e dele próprio
265 for i in range(2, int(number ** 0.5) + 1):
266 if number % i == 0:
267 return True # Encontrou divisor → é composto
268
269 return False # Não encontrou nenhum divisor → é primo
